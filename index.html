// 1. Install FusionAuth (Docker quickstart)
docker run -p 9011:9011 fusionauth/fusionauth-app

// 2. Setup API Key in FusionAuth UI (localhost:9011)

// 3. Node.js Example (using Express & FusionAuth SDK)
const express = require('express');
const { FusionAuthClient } = require('@fusionauth/typescript-client');
const app = express();
const client = new FusionAuthClient('YOUR_API_KEY', 'http://localhost:9011');

app.post('/register', async (req, res) => {
  try {
    const response = await client.register(null, { user: { email: req.body.email, password: req.body.password } });
    res.json(response);
  } catch (err) { res.status(500).json(err); }
});

app.listen(3000); // Run: node app.js
